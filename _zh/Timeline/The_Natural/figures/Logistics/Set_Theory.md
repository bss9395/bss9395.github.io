# 集合的运算性质
[空集] 𝟘 ≡ {e | e ≠ e}
[全集] 𝟙 ≡ {e | e = e}
[补集]    ¬S ≡ ∀e;[ e ∉ S]
[并集] S ∨ T ≡ ∀e;[[e ∈ S] ∨ [e ∈ T]]
[交集] S ∧ T ≡ ∀e;[[e ∈ S] ∧ [e ∈ T]]
[差集] S ⊖ T ≡ ∀e;[[e ∈ S] ∧ [e ∉ T]]
[对称] S ⊕ T ≡ ∀e;[[e ∈ S] ⊕ [e ∈ T]]
[相等] S = T ≡ ∀e;[[e ∈ S] ↔ [e ∈ T]]
[包含] S ⊆ T ≡ ∀e;[[e ∈ S] → [e ∈ T]]
[真含] S ⊂ T ≡ ∀e;∃t;[[[e ∈ S] → [e ∈ T]] ∧ ¬[[t ∈ T] → [t ∈ S]]]
[幂集] {P | P⊆S} ≡ ∀P;[P ⊆ S]
